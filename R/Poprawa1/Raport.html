df2_cont <- function(dataframe,name){
  #funkcja robi ramki danych do odpowiednich kontynentÃ³w
  dataframe %>% 
    filter(Continent == name,oct<15,apr>-5) %>% 
    mutate(Country = forcats::fct_reorder(Country,oct))
}


cont_plot <- function(dataframe,b=TRUE){
  #funkcja robi ploty dla wczeÅ›niejszych ramek kontynentÃ³w
  dataframe %>% 
    ggplot() +
    geom_segment( aes(x=Country, xend=Country, y=apr, yend=oct),size = 1.1, color = ifelse(dataframe$oct>dataframe$apr,"#51B86F","#FF0000")) +
    geom_point(aes(x=Country, y=apr), color="#2F0B36", size=1) +
    geom_point( aes(x=Country, y=oct,), color="#3387FF", size=1 ) +
    geom_hline(yintercept = 0,color="#a6a6a6")+
    coord_flip() +
    scale_x_discrete(guide = guide_axis(n.dodge = 2)) +
    scale_y_continuous(breaks=seq(-5,12.5,2.5), limits = c(-4,13.2)) +
    {if(b){ #zeby nie bylo x axis dla wszystkich kontynenow oprocz Oceanii na samym dole
      theme(axis.text.y = element_text(size = 9),
            axis.title.x=element_blank(),
            plot.margin = unit(c(0, 0, 0, 0), "cm"),
            axis.text.x = element_blank())
    }else{
      theme(axis.text.y = element_text(size = 9),
            axis.title.x=element_blank(),
            plot.margin = unit(c(0, 0.1, 0.3, 0.1), "cm"))
    }}
}
